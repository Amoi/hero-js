AfterMessage()

Source:
  • hero.js, line 202

Define the interceptor after @Message invoked.

import { AfterMessage, Component, Message } from 'hero-js';

@Component()
export class DecoratePage {

    @Message(function(data){
      return data.click && data.click === "login";
    })
    login(data) {
        console.log('Sending Request...');
    }

    @AfterMessage
    after(data) {
      console.log('Checking consistence of data...', data);
    }

}

Boot()

Callback method when page bootstrap. This method executed before @ViewWillAppear.
1. In Browser Environment: this method will executed every time when page loaded.
2. In Native Environment: this method execute one time only.

import { Boot, Component } from 'hero-js';

@Component()
export class DecoratePage {
 @Boot
 start(){
   console.log('Bootstrap Successfully!')
 }
}

Message(condition)

Source:
  • hero.js, line 490

Handle messages sent from NativeApp to JS. Given the condition, whether invoke the decorated function or not.

import { Component, Message } from 'hero-js';

@Component({
  view: {
            version:0,
            backgroundColor:'ffffff',
            nav:{
                navigationBarHiddenH5:true
            },
            views:[
                {
                    class:'DRTextField',
                    type:'phone',
                    theme:'green',
                    frame:{x:'15',r:'15',y:'115',h:'50'},
                    placeHolder:'手机号码',
                    name:'phone',
                    textFieldDidEditing:{name:'phone'},
                },
                {
                    class:'DRTextField',
                    theme:'green',
                    frame:{x:'15',r:'15',y:'178',h:'50'},
                    placeHolder:'密码',
                    secure:true,
                    name:'password',
                    drSecure:{'secure':true}, // 带小眼睛
                    textFieldDidEditing:{name:'password'},
                },
                {
                    class:'DRButton',
                    name:'loginBtn',
                    DRStyle:'B1',
                    enable:false,
                    frame:{x:'15',r:'15',y:'0',h:'44'},
                    yOffset:'password+50',
                    title:'登录',
                    // define the message sent from NativeApp to JS when the button click.
                    click:{click:'login'}
                },
            ]

  }
})
export class DecoratePage {

    // Receive the message from NativeApp
    @Message(function(data){
      // whether call the login method according to the result of this function
      return data.click && data.click === "login";
    })
    login(data) {
      console.log('Send request...');
    }
}
Parameters:
Name Type Description
condition function

condition function and the decorated function callback has one argument:
data: The message sent from NativeApp.

if condition is function, decorated function callback invoked when the result of condition equals to true.

@Message(function(data){
  // condition function
  function checkCondition(data){
     return true;
  }
  return checkCondition(data);
})
callback(data) {
  // decorated function
  console.log('Send request...');
}

if condition equals undefined, the decorated method callback will always invoked.

@Message()
callback(data){
 console.log('Received Data:', data);
}

Equals to

@Message(function(data){
 return true;
})
callback(data){
 console.log('Received Data:', data);
}
